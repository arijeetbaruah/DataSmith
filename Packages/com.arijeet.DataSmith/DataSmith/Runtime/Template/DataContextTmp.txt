using System;
using System.Collections.Generic;
using Baruah.DataSmith.Database;
using Newtonsoft.Json;

namespace Baruah.DataSmith
{
    /// <summary>
    /// Central runtime container for all <see cref="GameModel"/> instances.
    /// </summary>
    /// <remarks>
    /// <para>
    /// The <see cref="DataContext"/> acts as a lightweight in-memory database,
    /// responsible for:
    /// </para>
    /// <list type="bullet">
    /// <item><description>Creating and storing model instances</description></item>
    /// <item><description>Providing global access via <see cref="Get{T}"/></description></item>
    /// <item><description>Serializing and deserializing the entire model state</description></item>
    /// </list>
    ///
    /// <para>
    /// Models are registered through the <see cref="Types"/> array,
    /// which is typically auto-generated by DataSmith tooling.
    /// </para>
    ///
    /// <para>
    /// This class is conceptually similar to a database context
    /// or service locator for game data.
    /// </para>
    /// </remarks>
    public static class DataContext
    {
        /// <summary>
        /// Internal registry mapping model types to their instances.
        /// </summary>
        private static Dictionary<System.Type, GameModel> _models = new();
        
        /// <summary>
        /// Gets a value indicating whether the context has been initialized.
        /// </summary>
        /// <remarks>
        /// Initialization occurs automatically when accessing models,
        /// but can also be triggered manually via <see cref="Initialize"/>.
        /// </remarks>
        public static bool IsInitialized { get; private set; }
        
        /// <summary>
        /// Gets the list of model types that should be created and managed
        /// by this context.
        /// </summary>
        /// <remarks>
        /// This array is typically generated automatically by DataSmith.
        /// Only types deriving from <see cref="GameModel"/> will be instantiated.
        /// </remarks>
        public static readonly Type[] Types =
        {
{{CONTEXT_TYPE}}
        };
        
        public static IDatabase Database
        {
            get
            {
                if (_database != null)
                {
                    return _database;
                }
                
                throw new System.Exception("Database is not initialized");
            }
        }
        
        private static IDatabase _database;

        /// <summary>
        /// Initializes the context and creates instances for all registered models.
        /// </summary>
        /// <remarks>
        /// <para>
        /// This method is safe to call multiple times; initialization
        /// will only occur once.
        /// </para>
        ///
        /// <para>
        /// Model instances are created using <see cref="Activator.CreateInstance(Type)"/>,
        /// so each model type must provide a parameterless constructor.
        /// </para>
        /// </remarks>
        public static void Initialize()
        {
            if (IsInitialized)
                return;

            foreach (var type in Types)
            {
                if (!typeof(GameModel).IsAssignableFrom(type))
                    continue;

                var instance =
                    (GameModel)Activator.CreateInstance(type);

                _models[type] = instance;
            }

            IsInitialized = true;
        }

        /// <summary>
        /// Gets the model instance of the specified type.
        /// </summary>
        /// <typeparam name="T">
        /// The type of the model to retrieve.
        /// </typeparam>
        /// <returns>
        /// The initialized model instance.
        /// </returns>
        /// <exception cref="Exception">
        /// Thrown if the requested model type is not registered
        /// in <see cref="Types"/>.
        /// </exception>
        /// <remarks>
        /// <para>
        /// The context will automatically initialize itself if needed.
        /// </para>
        ///
        /// <para>
        /// This method provides global, strongly-typed access to models.
        /// </para>
        /// </remarks>
        public static T Get<T>() where T : GameModel
        {
            Initialize();

            if (_models.TryGetValue(typeof(T), out var model))
                return (T)model;

            throw new Exception($"Model not registered: {typeof(T).Name}");
        }

        /// <summary>
        /// Serializes all registered models into a JSON string.
        /// </summary>
        /// <returns>
        /// A JSON representation of the entire context state.
        /// </returns>
        /// <remarks>
        /// <para>
        /// Uses Newtonsoft.Json for serialization.
        /// </para>
        ///
        /// <para>
        /// This method is suitable for saving game state to disk,
        /// cloud storage, or network transmission.
        /// </para>
        /// </remarks>
        public static string SerializeAll()
        {
            Initialize();
            
            return JsonConvert.SerializeObject(_models);
        }

        /// <summary>
        /// Restores all models from a JSON string produced by
        /// <see cref="SerializeAll"/>.
        /// </summary>
        /// <param name="json">
        /// The serialized context data.
        /// </param>
        /// <remarks>
        /// <para>
        /// Existing model instances will be replaced by the deserialized ones.
        /// </para>
        ///
        /// <para>
        /// The context will be initialized automatically if necessary.
        /// </para>
        ///
        /// <para>
        /// Ensure the serialized data matches the current model definitions
        /// to avoid runtime errors.
        /// </para>
        /// </remarks>
        public static void DeserializeAll(string json)
        {
            Initialize();

            _models = JsonConvert.DeserializeObject<Dictionary<System.Type, GameModel>>(json);
        }
    }
}
